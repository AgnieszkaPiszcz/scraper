def create_csv(cars)
    begin
        csv = CSV.open("output/cars.csv", "wb") do |csv|
            csv << cars.first.keys 
            cars.each do |car|
            csv << car.values
            end
        end   
    rescue CSV::Error => e
        puts "CSV handling error: #{e.class}. Try again"
        exit(1)
    end

end
